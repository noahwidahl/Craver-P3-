-- Creating the Roles Table with a Unique Constraint on RoleDescription
CREATE TABLE Roles (
    RoleID INT AUTO_INCREMENT PRIMARY KEY,
    RoleDescription VARCHAR(100) UNIQUE NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Creating the User Table
CREATE TABLE User (
    UserID INT AUTO_INCREMENT PRIMARY KEY,
    UserName VARCHAR(100) UNIQUE NOT NULL,
    Password VARCHAR(25) NOT NULL,
    Email VARCHAR(100) UNIQUE NOT NULL,
    Address VARCHAR(100),
    PostNr VARCHAR(4),
    PostBy VARCHAR(100),
    UserRoleID INT,
    AddDate DATETIME DEFAULT CURRENT_TIMESTAMP,
    ModifiedDate DATETIME ON UPDATE CURRENT_TIMESTAMP,
    LastSeen DATETIME ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (UserRoleID) REFERENCES Roles(RoleID)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Junction table for FoodItemIngredients, 
CREATE TABLE craveconnect.FoodItemIngredients (
    FoodItemID INT,
    IngredientID INT,
    PRIMARY KEY (FoodItemID, IngredientID),
    FOREIGN KEY (FoodItemID) REFERENCES FoodItem(FoodItemID) ON DELETE CASCADE ON UPDATE CASCADE,
    FOREIGN KEY (IngredientID) REFERENCES Ingredients(IngredientID) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Creating a view to join user table and roles table in a combined table
create view v_UserAndRoles as
SELECT UserID, UserName, PassWord, Email, Address, PostNr, PostBy, UserRoleID, RoleDescription, LastSeen  FROM craveconnect.User T1
inner join craveconnect.Roles T2 on T1.UserRoleID = T2.RoleID;

-- stored procedure to handle the sequence of deleting a user profile
DELIMITER //
CREATE PROCEDURE craveconnect.sp_DeleteUserProcedure(IN p_UserID INT)
BEGIN
    -- First we delete the ratings of the user
    DELETE FROM craveconnect.Rating WHERE UserID = p_UserID;

    -- Then we delete the user himself
    DELETE FROM craveconnect.User WHERE UserID = p_UserID;
END //
DELIMITER ;

-- stored procedure to handle the sequence of deleting a FoodItem
DELIMITER //
CREATE PROCEDURE craveconnect.sp_DeleteFoodItem(IN p_FoodItemID INT)
BEGIN
    -- First we delete the ratings of the foodItem
    delete from craveconnect.Rating where FoodItemID = p_FoodItemID;

    -- Then we delete the foodItem
    delete from craveconnect.FoodItem where FoodItemID = p_FoodItemID;
END //
DELIMITER ;

-- CALL craveconnect.sp_DeleteFoodItem(5);
-- Stored procedure to handle the sequence of deleting a Foodsupplier
DELIMITER //
CREATE PROCEDURE craveconnect.sp_DeleteFoodsupplier(IN p_foodsupplierID INT)
BEGIN
	-- To disable safe mode
	SET SQL_SAFE_UPDATES = 0;
	-- Deleting rating of the foodsuppliers foodItems
	delete from craveconnect.Rating where FoodItemID in (select FoodItemID from craveconnect.FoodItem where FoodSupplierID = p_foodsupplierID);
	-- Deleting the food items owned by the foodsupplier
	delete from craveconnect.FoodItem where FoodSupplierID = p_foodsupplierID;
	-- Deleting the foodsupplier themselv
	delete from craveconnect.Foodsupplier where FoodSupplierID = p_foodsupplierID;
	-- To enable safe mode
	SET SQL_SAFE_UPDATES = 1;
END //
DELIMITER ;
